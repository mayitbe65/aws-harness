━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Q: 前端显示 400 Bad Request
A: Vite 配置已修复，重启 npm run dev

Q: 后端无法启动
A: 确认虚拟环境激活：source venv/bin/activate
   然后：pip install -r requirements.txt

Q: 前端显示空白
A: 打开浏览器 DevTools (F12) → Console，查看错误

Q: 无法连接到 API
A: 确保后端在运行：curl http://localhost:8000/docs
